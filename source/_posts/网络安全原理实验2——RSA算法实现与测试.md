def f(r, x, n):
    """
    计算 x 的 r 次方模 n
    """
    a = x
    b = r
    c = 1
    while b != 0:
        while b % 2 == 0:
            b = b // 2
            a = (a * a) % n
        if b % 2 != 0:
            b = b - 1
            c = (c * a) % n
    return c


def Enclid(n, d):
    """
    计算模逆运算
    """
    a = n
    b = d
    t = 0
    v = 1
    w = 0
    while True:
        r = a % b
        if r != 0:
            q = (a - r) // b
            a = b
            b = r
            w = v
            v = t - q * v
            t = w
        else:
            break
    if b == 1:
        if v >= 0:
            return v
        else:
            return n + v
    else:
        return 0


def gcd(a, b):
    """
    计算最大公约数
    """
    r = a % b
    while True:
        r = a % b
        if r != 0:
            a = b
            b = r
        else:
            break
    if b == 1:
        return 1
    else:
        return b
